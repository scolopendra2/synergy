age = input('Введите свой возраст: ')
if not age.isdigit():
    print('ERROR! Это не число')
else:
    city = input('Введите свой город: ')
    name = input('Введите своё имя: ')
    print('Доступные варианты ответа: +, -')
    together = input('С вами есть сопровождающий: ')
    if together not in ['+', '-']:
        print('ERROR! Нет такого варианта ответа.')
    else:
        print(f'Здравстуйте {name}!\nДля города {city} в кинотеатре:')
        if int(age) < 12:
            print('Билет бесплатный')
        elif 12 <= int(age) <= 17:
            print('Билет со скидкой')
        else:
            print'Полная стоимость билета'
